		this.isRetrievingArticles = true;
 
		this.articleError = new Message();
 
		// init uploader
		const uploaderOptions: FileUploaderOptions = {
			url: 'https://api.cloudinary.com/v1_1/' + HelperService.cloudName + '/image/upload',
			isHTML5: true,
			removeAfterUpload: true,

		this.uploader = new FileUploader(uploaderOptions);

	ngOnInit() {
		// get all articles (self and followings)
		this.articleService.getAllCurrentUserFeeds()

			.catch(err => {
				this.isRetrievingArticles = false;
 
				this.displayError(err);
			});
 
		this.initImageForm();
	}
 
	// init image form and set listeners
	private initImageForm(): void {
		// overwrite build avatar form method
		this.uploader.onBuildItemForm = (fileItem: any, form: FormData): any => {
			form.append('file', fileItem);
 
			HelperService.generateFormData(form, { folder: "article" });
 
			fileItem.withCredentials = false;
 
			return { fileItem, form };
		};
 
		// after uploading
		this.uploader.onCompleteItem = (file: any, res: string, status: number, headers: ParsedResponseHeaders) => {
			this.imageEditor.isEditing = false;
 
			const resObj = JSON.parse(res);
 
			// handle error
			if (resObj.error) {
				this.newArticleEditor.isHandling = false;
 
				return this.displayError("Fail to upload image.");
			}
 
			this.post(resObj.url);
		};
 
		// update upload progress
		this.uploader.onProgressItem = (file: any, progress: any) => this.uploadProgress = progress;
	}
 
	// show new avatar info
	private showImageInfo(): void {
		this.imageEditor.value = this.uploader.queue[0].file.name;
		this.imageEditor.isEditing = true;
	}
 
	// upload the new image
	private uploadImage(): void {
		this.uploadProgress = 0;
 
		this.uploader.queue[0].upload();
	}
 
	// clear the selected image
	private clearImage(): void {
		this.imageEditor.isEditing = false;
 
		this.uploader.clearQueue();
	}
 
	// post the article
	private post(imageUrl: string): void {
		this.articleService.postArticle({
			text: this.newArticleEditor.value,
			image: imageUrl
		})
			.then(article => {
				// add the new article to the head of the articles list
				this.articles.splice(0, 0, article);
				this.activeArticles.splice(0, 0, article);
 
				this.newArticleEditor.value = "";
				this.newArticleEditor.isHandling = false;
				this.articleError.isShowing = false;
			})
			.catch(err => {
				this.newArticleEditor.isHandling = false;
 
				this.displayError(err);
			});
	}
 
	// handle retrieved articles
	private setArticles(articles: any[]): void {
		this.articles = articles.concat();
 
		// sort by time
		this.articles.sort((a1, a2) => a1.createTime > a2.createTime ? -1 : 1);
 
		// initially, all articles will be displayed
		this.activeArticles = this.articles.concat();
 
		this.isRetrievingArticles = false;
	}

	private filter(): void {
		this.activeArticles = this.articles.filter(article =>
			article.author.indexOf(this.articlesFilterEditor.value) != -1
				|| article.text.indexOf(this.articlesFilterEditor.value) != -1);
	}
 
	// remove articles by username (called by parent)
	public removeByUsername(username: string): void {
		const articles = this.articles.filter(article => article.author !== username);
 
		this.setArticles(articles);
	}
 
	// add articles by username (called by parent)
	public addByUsername(username: string): void {
		this.isRetrievingArticles = true;
 
		this.articleService.getAllArticlesByUsername(username)
			.then(res => {
				// concat articles
				const articles = this.articles.concat(res);
 
				this.setArticles(articles);
 
				this.articleError.isShowing = false;
			})
			.catch(err => {
				this.isRetrievingArticles = false;
 
				this.displayError(err);
			});
	}
 
	private displayError(err): void {
		this.articleError.isShowing = true;
		this.articleError.message = err;
	}
 
}
