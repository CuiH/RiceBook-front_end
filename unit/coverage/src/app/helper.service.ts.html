import { Injectable } from '@angular/core';
import { Router } from '@angular/router';
 
import { LocalService } from './local.service';
 
import api from 'sha1';
 
 
@Injectable()
export class HelperService {
	// heroku
	public static remoteUrl = "https://hw6-back-end.herokuapp.com/";
	// public static remoteUrl = "http://localhost:3000/";
 
	// cloudinary
	public static cloudName = "hcui95";
	public static api_key = "359374882927652";
	public static api_secret = "WsibNv_x-8Aa-n2ErOCwksp6vHk";
 
 
	constructor(
		private router: Router,
		private localService: LocalService
	) { }
 
	public static generateEditAvatarUrl(url: string, type: string): string {
		const parts = url.split('upload');
 
		switch(type) {
			case 'F':
				return parts[0] + "upload/" + "c_fill,h_259,w_220" + parts[1];
			case 'B':
				return parts[0] + "upload/" + "c_fill,h_325,w_275" + parts[1];
			case 'P':
				return parts[0] + "upload/" + "c_fill,h_170,w_170" + parts[1];
			case 'C':
				return parts[0] + "upload/" + "c_fill,h_100,w_100" + parts[1];
			case 'A':
				return parts[0] + "upload/" + "c_fill,h_100,w_100" + parts[1];
		}
	}
 
	public generateErrorHandler(): any {
		return err => {
			if (err.status == 401) {
				this.localService.clearLoggedInUsername();
 
				this.router.navigate(['/landing'], {queryParams: {type: 'log_in'}});
			} else {
				return Promise.reject(err.json().message);
			}
		}
	}
 
	// append data to form, keys in alphabetic order
	public static generateFormData(form: FormData, values): any {
		form.append('api_key', HelperService.api_key);
 
		let strs = [];
 
		// add custom data
		Object.keys(values).forEach(key => {
			form.append(key, values[key]);
			strs.push(key + "=" + values[key]);
		});
 
		// add time stamp
		const timestamp = "" + Math.floor(new Date().getTime() / 1000);
		form.append("timestamp", timestamp);
		strs.push("timestamp=" + timestamp);
 
		form.append('signature', api(strs.join('&') + HelperService.api_secret));
	}
 
}
